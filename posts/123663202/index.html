int Fiboncci(int n)
{
  int[] result=new int[] { 0, 1 };
  if (n < 2)
     return result[n];

  int m = 1;
  int n = 0;
  int k = 0;
  for (int i = 2; i <= n; i++)
  {
     k = m + n;
     n = m;
     m = k;
  }

  return k;
}
