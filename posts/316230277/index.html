йҳ…иҜ»ж—¶й•ҝ: 5 еҲҶй’ҹ
AI ж‘ҳиҰҒ
EasyAR е°қйІңзі»еҲ—ж•ҷзЁӢд»Ӣз»ҚдәҶеҰӮдҪ•еңЁUnityдёӯе®һзҺ°и§Ҷйў‘ж’ӯж”ҫеҠҹиғҪпјҢдё»иҰҒйҖҡиҝҮVideoPlayerBehaviourзұ»жқҘж”ҜжҢҒи§Ҷйў‘ж’ӯж”ҫгҖӮдҪңиҖ…жҸҗеҲ°дәҶи§Ҷйў‘ж’ӯж”ҫеҷЁз»„д»¶зҡ„и®ҫзҪ®еҸӮж•°д»ҘеҸҠеҰӮдҪ•еңЁImageTargetдёӢж·»еҠ VideoPlayerBehaviourзҡ„еӯҗзү©дҪ“жқҘе®һзҺ°VideoTargetгҖӮжӯӨеӨ–пјҢдҪңиҖ…иҝҳеұ•зӨәдәҶдёҖдёӘиҮӘе®ҡд№үзҡ„VideoTargetBaseBehaviourзұ»пјҢйҖҡиҝҮеҠЁжҖҒеҲӣе»әи§Ҷйў‘з»„д»¶жқҘеўһејәImageTargetпјҢдҪҝеҫ—еҲ¶дҪңиҜҶеҲ«еӣҫж—¶жӣҙдё“жіЁдё”ж–№дҫҝгҖӮж•ҷзЁӢдёӯжҸҗеҲ°дәҶеҜ№32дҪҚж“ҚдҪңзі»з»ҹдёӢи§Ҷйў‘ж’ӯж”ҫзҡ„е…је®№жҖ§й—®йўҳпјҢе№¶з»ҷеҮәдәҶзӣёе…ід»Јз ҒзӨәдҫӢе’Ңй…ҚзҪ®жӯҘйӘӨгҖӮ

еҗ„дҪҚжңӢеҸӢеӨ§е®¶еҘҪпјҢж¬ўиҝҺеӨ§е®¶е…іжіЁжҲ‘зҡ„еҚҡе®ўпјҢжҲ‘жҳҜз§Ұе…ғеҹ№пјҢжҲ‘зҡ„еҚҡе®ўең°еқҖжҳҜhttp://qinyuanpei.comгҖӮеҲ°зҺ°еңЁдёәжӯўпјҢжҲ‘д»¬еҜ№ EasyAR дёӯзҡ„ ImageTarget еҹәжң¬дёҠеҸҜд»ҘиҜҙжҳҜй©ҫиҪ»е°ұзҶҹдәҶпјҢеӣ жӯӨжҲ‘д»¬иҝҷдёӘзі»еҲ—ж•ҷзЁӢеҸҜд»ҘиҜҙжҳҜжҺҘиҝ‘е°ҫеЈ°дәҶгҖӮеҚҡдё»з¬¬дёҖж¬ЎжҺҘи§Ұ AR иҝҷдёӘжҰӮеҝөжҳҜеңЁеӨ§еӯҰж—¶еҖҷиҜ»еҲ°дёҖжң¬и®Іи§Ји®Ўз®—жңәеӣҫеҪўи§Ҷи§үзҡ„д№ҰзұҚйҮҢпјҢзӣёеҜ№ VR жҠҖжңҜзӣ®еүҚеҚҺиҖҢдёҚе®һзҡ„еёӮеңәзҺ°зҠ¶пјҢAR жҠҖжңҜд»Һе®һз”ЁжҖ§е’ҢжҲҗзҶҹеәҰйғҪиғҪеҫ—еҲ°иҫғеҘҪзҡ„дҝқиҜҒгҖӮеҸҜжҳҜеӨ§е®¶йғҪжё…жҘҡиҝҷдәӣжҠҖжңҜиғҢеҗҺйғҪжҳҜе»әз«ӢеңЁеӨҚжқӮиҖҢй«ҳж·ұзҡ„еӣҫеҪўеӯҰз®—жі•зҡ„еҹәзЎҖдёҠзҡ„пјҢеҰӮжһңжғіеӯҰд№  AR жҠҖжңҜиҜ·еӣһеҪ’и®Ўз®—жңәеӣҫеҪўеӯҰзҡ„жң¬жәҗпјҢиҝҷе°ұе’ҢеӯҰд№ жёёжҲҸжҠҖжңҜиҰҒиҝҪеҜ»еҸҜзј–зЁӢжёІжҹ“з®ЎзәҝжҳҜдёҖж ·зҡ„пјҢжүҖд»ҘиҝҷдёӘзі»еҲ—е®Ңе…ЁжҳҜеҚҡдё»дёӘдәәзҡ„е…ҙи¶ЈдҪҝз„¶пјҢеёҢжңӣдәҶи§ЈиҝҷдёӘжҠҖжңҜзҡ„еҸҜд»ҘиҝӣиЎҢжӣҙеҠ ж·ұе…Ҙзҡ„жҺўзҙўгҖӮиҝҷж¬ЎжҲ‘д»¬жқҘиҜҙиҜҙ VideoTarget еҰӮдҪ•е®һзҺ°еҗ§пјҒ

EasyAR дёӯеҜ№и§Ҷйў‘зҡ„ж”ҜжҢҒ

зӣ®еүҚ EasyAR еҜ№и§Ҷйў‘зҡ„ж”ҜжҢҒдё»иҰҒжҳҜйҖҡиҝҮ VideoPlayerBehaviour иҝҷдёӘзұ»пјҢиҝҷдёӘзұ»з»§жүҝиҮӘдёҖдёӘеҹәзұ» VideoPlayerBaseBehaviourгҖӮжҲ‘д»¬еҸҜд»Ҙе°Ҷе…¶зҗҶи§ЈдёәдёҖдёӘи§Ҷйў‘ж’ӯж”ҫеҷЁз»„д»¶пјҢеҸӘиҰҒжҲ‘д»¬е°ҶиҝҷдёӘз»„д»¶ж·»еҠ еҲ°дёҖдёӘ GameObject дёҠпјҢз„¶еҗҺз®ҖеҚ•еЎ«еҶҷдёӢеҸӮж•°е°ұеҸҜд»ҘдәҶгҖӮеҸҜжҳҜиҝҷдёӘз»„д»¶еҚҡдё»еңЁ 32 дҪҚж“ҚдҪңзі»з»ҹдёӢе№¶жІЎжңүзңӢеҲ°е®һйҷ…зҡ„ж•ҲжһңпјҢиҷҪз„¶иҜҙйғҪеҲ°дәҶ 2015 е№ҙдәҶ 64 дҪҚж“ҚдҪңзі»з»ҹзӣёеҜ№жқҘиҜҙжӣҙдёәжҷ®еҸҠдәҶпјҢеҸҜжҳҜжҲ‘и§үеҫ—ж”ҜжҢҒдёҚж”ҜжҢҒ 32 дҪҚж“ҚдҪңзі»з»ҹжӣҙеӨҡзҡ„дҪ“зҺ°зҡ„жҳҜдёҖе®¶е…¬еҸёеҒҡдә§е“Ғзҡ„жҖҒеәҰгҖӮж—ўз„¶жҡӮж—¶жІЎжңүеҠһжі•зңӢеҲ°иҝҷйҮҢзҡ„е…·дҪ“ж•ҲжһңпјҢжҲ‘д»¬жң¬зқҖеӯҰд№ зҡ„жҖҒеәҰеҜ№иҝҷдёӘз»„д»¶жңүжүҖдәҶи§Је°ұжҳҜдәҶгҖӮдёӢйқўжҳҜиҝҷдёӘз»„д»¶зҡ„дёҖеј жҲӘеӣҫпјҡ

д»ҺеӣҫдёӯжҲ‘д»¬еҸҜд»ҘзңӢеҲ°иҝҷдёӘз»„д»¶зӣёе…іеҸӮж•°зҡ„и®ҫзҪ®пјҢиҝҷйҮҢйҖүеҸ–зҡ„и§Ҷйў‘иө„жәҗжҳҜ StreamingAssets зӣ®еҪ•дёӢзҡ„ video.mp4 иҝҷдёӘж–Үд»¶пјҢи§Ҷйў‘иө„жәҗзҡ„ Stroge еҗҢж ·ж”ҜжҢҒ AppгҖҒAssetsгҖҒAbsolute иҝҷдёүз§Қзұ»еһӢпјҢе’ҢеӣҫзүҮиө„жәҗзҡ„ Stroge жҳҜдёҖж ·зҡ„пјҢе…ідәҺиҝҷдёүз§Қзұ»еһӢзҡ„иө„жәҗи·Ҝеҫ„зҡ„й—®йўҳпјҢжҲ‘иҝҷйҮҢдёҚжғіеҶҚйҮҚеӨҚиҜҙдәҶпјҢиҝҷдёӘзңӢзңӢж–ҮжЎЈе°ұзҹҘйҒ“дәҶгҖӮе…¶ж¬Ўдјҡж¶үеҸҠеҲ°и§Ҷйў‘ж’ӯж”ҫж–№ејҸе’Ңи§Ҷйў‘зј©ж”ҫзҡ„зӣёе…іеҸӮж•°пјҢиҝҷдәӣеҹәжң¬дёҠжІЎд»Җд№ҲзҗҶи§ЈдёҠзҡ„йҡҫеәҰпјҢеӨ§е®¶еҜ№з…§зқҖж–ҮжЎЈеҸҚеӨҚе°қиҜ•е°ұзҹҘйҒ“еҗ„иҮӘзҡ„з”ЁйҖ”дәҶгҖӮеҚҡдё»иҝҷйҮҢдёҚеӨӘзҗҶи§Ј EasyAR дёәд»Җд№ҲдёҚйҮҮз”Ё MovieTexture жҲ–иҖ… Unity3D дёӯй’ҲеҜ№и§Ҷйў‘ж’ӯж”ҫжҸҗдҫӣзҡ„зӣёе…іжҸ’д»¶пјҢеӣ дёә VideoTarget жң¬иҙЁдёҠе°ұжҳҜжҠҠдёүз»ҙжЁЎеһӢжҚўжҲҗдәҶеҸҜд»Ҙж’ӯж”ҫзҡ„и§Ҷйў‘иҖҢе·ІпјҢжүҖд»ҘеӨ§е®¶еңЁеүҚйқўж–Үз« зҡ„еҹәзЎҖдёҠеҲӣе»әдёҖдёӘ ImageTarget з„¶еҗҺеҶҚе…¶дёӢйқўж”ҫзҪ®дёҖдёӘйҷ„еҠ дәҶ VideoPlayerBehaviour зҡ„зҡ„еӯҗзү©дҪ“е°ұеҸҜд»ҘдәҶгҖӮе®ҳж–№зҡ„зӨәдҫӢйЎ№зӣ®дёӯжҸҗдҫӣдәҶдёӨз§Қж–№ејҸзҡ„ VideoTarget еҲӣе»әж–№ејҸпјҢеҚіжүӢеҠЁеҲӣе»әе’ҢеҠЁжҖҒеҲӣе»әгҖӮжүӢеҠЁеҲӣе»әеҚіжҲ‘д»¬иҝҷйҮҢжҸҗеҲ°зҡ„иҝҷз§Қж–№ејҸпјҢиҖҢеҠЁжҖҒеҲӣе»әеҲҷжҳҜз”ұзЁӢеәҸеңЁиҝҗиЎҢж—¶жңҹй—ҙеҲӣе»әгҖӮиҝҷдёӨз§Қж–№ејҸжң¬иҙЁдёҠжІЎжңүд»Җд№ҲдёҚеҗҢпјҢйңҖиҰҒжіЁж„Ҹзҡ„жҳҜ VideoPlayerBehaviour жңүдёҖдёӘ EnableAutoPlay зҡ„йҖүйЎ№пјҢиҜҘйҖүйЎ№иў«йҖүдёӯеҗҺдјҡеҗҜз”ЁиҮӘеҠЁж’ӯж”ҫпјҢеҚіеҪ“иҜҶеҲ«еӣҫиў«иҜҶеҲ«еҗҺиҮӘеҠЁж’ӯж”ҫи§Ҷйў‘гҖҒиҜҶеҲ«еӣҫжңӘиў«иҜҶеҲ«еҲҷжҡӮеҒңж’ӯж”ҫи§Ҷйў‘гҖӮеҰӮжһңиҝҷдёӘйҖүйЎ№жІЎжңүиў«йҖүдёӯпјҢжҲ‘д»¬йңҖиҰҒеңЁ ITargetEventHandle жҺҘеҸЈдёӯеҠЁжүӢжқҘе®һзҺ°гҖӮ

еўһејә ImageTarget

иҝҷдёӘеўһејә ImageTarget жҳҜжҢҮеңЁ ImageTarget зҡ„еҹәзЎҖдёҠиһҚе…Ҙ VideoPlayerBehaviour зҡ„еҠҹиғҪпјҢеӣ дёәжҢүз…§е®ҳж–№зҡ„зӨәдҫӢжқҘиҖғиҷ‘пјҢиҝҷдёӨйғЁеҲҶеҠҹиғҪжҳҜзӢ¬з«Ӣзҡ„пјҢеҚҡдё»еёҢжңӣи®©еӨ§е®¶еңЁеҲ¶дҪңиҜҶеҲ«еӣҫзҡ„ж—¶еҖҷе®Ңе…Ёеҝҳи®°еҢәеҲ« ImageTarget е’Ң VideoTargetпјҢиҝҷж ·жҲ‘д»¬еҸҜд»Ҙжӣҙдёәдё“жіЁең°еҲ¶дҪңиҜҶеҲ«еӣҫпјҢеӣ дёәи§Ҷйў‘з»„д»¶е°ұеҸӘжҳҜи®ҫзҪ®еҸӮж•°иҝҷдёҖд»¶дәӢжғ…пјҢе®Ңе…ЁеҸҜд»ҘдёҖж¬ЎжҖ§жҗһе®ҡпјҢжүҖд»ҘжҲ‘д»¬йҰ–е…ҲжқҘе®ҡд№үдёҖдёӘ VideoTargetBaseBehaviour зұ»пјҢдёҖиө·жқҘзңӢд»Јз Ғпјҡ

using UnityEngine;
using System.Collections;
using EasyAR;

public class VideoTargetBaseBehaviour : ImageTargetBehaviour,ITargetEventHandler
{
    /// <summary>
    /// и§Ҷйў‘ж’ӯж”ҫжЁЎеқ—
    /// </summary>
    private VideoPlayerBehaviour videoPlayer;

    /// <summary>
    /// и§Ҷйў‘ж–Үд»¶и·Ҝеҫ„
    /// </summary>
    public string VideoPath;

    /// <summary>
    /// жҳҜеҗҰиҮӘеҠЁж’ӯж”ҫи§Ҷйў‘
    /// </summary>
    public bool VideoEnableAutoPlay = true;

    /// <summary>
    /// жҳҜеҗҰе…Ғи®ёи§Ҷйў‘еҫӘзҺҜ
    /// </summary>
    public bool VideoEnableLoop = true;

    /// <summary>
    /// и§Ҷйў‘зұ»еһӢ
    /// </summary>
    public VideoPlayer.VideoType VideoType = VideoPlayer.VideoType.TransparentSideBySide;

    /// <summary>
    /// и§Ҷйў‘иө„жәҗзұ»еһӢ
    /// </summary>
    public StorageType VideoStorage = StorageType.Assets;


    /// <summary>
    /// и§Ҷйў‘жҳҜеҗҰеҠ иҪҪ
    /// </summary>
    private bool isVideoLoaded;

    protected override void Start()
    {
        //еңЁStartж–№жі•дёӯеҠ иҪҪи§Ҷйў‘гҖҒйҡҗи—ҸжЁЎеһӢ
        base.Start();
        LoadVideo();
        HideObjects(transform);
    }

    /// <summary>
    /// еҠ иҪҪи§Ҷйў‘
    /// </summary>
    private void LoadVideo()
    {
        //еҲӣе»әеӯҗзү©дҪ“VideoObjectе№¶дёәе…¶ж·»еҠ и§Ҷйў‘з»„д»¶
        GameObject VideoObject = new GameObject("VideoObject");
        videoPlayer = VideoObject.AddComponent<VideoPlayerBehaviour>();
        VideoObject.transform.SetParent(transform);
        VideoObject.transform.localPosition = Vector3.zero;
        VideoObject.transform.localRotation = Quaternion.identity;
        VideoObject.transform.localScale = Vector3.one;

        //и®ҫзҪ®и§Ҷйў‘з»„д»¶зӣёе…іеҸӮж•°
        videoPlayer.Storage = VideoStorage;
        videoPlayer.Path = VideoPath;
        videoPlayer.EnableAutoPlay = VideoEnableAutoPlay;
        videoPlayer.EnableLoop = VideoEnableLoop;
        videoPlayer.Type = VideoType;
        videoPlayer.VideoReadyEvent+=videoPlayer_VideoReadyEvent;
        videoPlayer.VideoReachEndEvent+=videoPlayer_VideoReachEndEvent;
        videoPlayer.VideoErrorEvent+=videoPlayer_VideoErrorEvent;
        videoPlayer.Open();

        videoPlayer.Play();
    }

    #region и§Ҷйў‘з»„д»¶зӣёе…ідәӢд»¶е®ҡд№ү

    public virtual void videoPlayer_VideoErrorEvent(object sender, System.EventArgs e)
    {
        
    }

    public virtual void videoPlayer_VideoReachEndEvent(object sender, System.EventArgs e)
    {
        
    }

    
    public virtual void videoPlayer_VideoReadyEvent(object sender, System.EventArgs e)
    {
        
    }

    #endregion

    /// <summary>
    /// йҡҗи—ҸжЁЎеһӢзҡ„ж–№жі•
    /// </summary>
    /// <param name="trans">иҰҒйҡҗи—Ҹзҡ„Transform</param>
    void HideObjects(Transform trans)
    {
        for (int i = 0; i < trans.childCount; ++i)
            HideObjects(trans.GetChild(i));
        if (transform != trans)
            gameObject.SetActive(false);
    }

    /// <summary>
    /// жҳҫзӨәжЁЎеһӢзҡ„ж–№жі•
    /// </summary>
    /// <param name="trans">иҰҒжҳҫзӨәзҡ„Transform</param>
    public void ShowObjects(Transform trans)
    {
        for (int i = 0; i < trans.childCount; ++i)
            ShowObjects(trans.GetChild(i));
        if (transform != trans)
            gameObject.SetActive(true);
    }

    /// <summary>
    /// е®һзҺ°ITargetEventHandlerжҺҘеҸЈдёӯзҡ„OnTargetFoundж–№жі•
    /// </summary>
    /// <param name="target">иҜҶеҲ«зӣ®ж Ү</param>
    void ITargetEventHandler.OnTargetFound(Target target)
    {
        if (videoPlayer)
            videoPlayer.Play();
        ShowObjects(transform);
    }

    /// <summary>
    /// е®һзҺ°ITargetEventHandlerжҺҘеҸЈдёӯзҡ„OnTargetLostж–№жі•
    /// </summary>
    /// <param name="target">иҜҶеҲ«зӣ®ж Ү</param>
    void ITargetEventHandler.OnTargetLost(Target target)
    {
        if (videoPlayer)
            videoPlayer.Pause();
        HideObjects(transform);
    }

    /// <summary>
    /// е®һзҺ°ITargetEventHandlerжҺҘеҸЈдёӯзҡ„OnTargetLoadж–№жі•
    /// </summary>
    /// <param name="target">иҜҶеҲ«зӣ®ж Ү</param>
    void ITargetEventHandler.OnTargetLoad(Target target, bool status)
    {

    }

    /// <summary>
    /// е®һзҺ°ITargetEventHandlerжҺҘеҸЈдёӯзҡ„OnTargetUnloadж–№жі•
    /// </summary>
    /// <param name="target">иҜҶеҲ«зӣ®ж Ү</param>
    void ITargetEventHandler.OnTargetUnload(Target target, bool status)
    {

    }
}
